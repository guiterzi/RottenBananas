-- Ver todos os atores
SELECT a.pessoa_id, p.nome 
FROM "Ator" a
JOIN "Pessoa" p ON a.pessoa_id = p.pessoa_id;

-- Ver todos os filmes
SELECT * FROM "Filme";

-- Ver todos os usuarios
SELECT * FROM "Usuario";

-- Ver atores de um filme específico
SELECT p.nome, af.papel
FROM "Ator_Filme" af
JOIN "Pessoa" p ON af.ator_id = p.pessoa_id
JOIN "Filme" f ON af.filme_id = f.filme_id
WHERE f.nome = 'A Lista de Schindler';

-- Ver avaliações de um filme
SELECT u.nome, a.nota, a.comentario
FROM "Avaliacao" a
JOIN "Usuario" u ON a.usuario_id = u.usuario_id
JOIN "Filme" f ON a.filme_id = f.filme_id
WHERE f.nome = 'O Diabo Veste Prada';
